def initialize(**definition, &block)
  traits = definition.extract!(*CORE_TRAITS)
  traits.each { |trait, value| public_send "#{trait}=", value }
  
  if @original_race = definition.delete(:subrace_of)
    singleton_class.prepend SubraceRules
  end
  
  super(&block)
end
